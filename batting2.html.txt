// ── スコア計算（0〜100） ──────────────────────────────
function calcScore(k,v){
  if(!v&&v!==0)return null;
  const n=parseFloat(v),m=MLB[k]?.val;
  if(isNaN(n)||!m)return null;
  // バット角度はマイナス値基準：MLB平均-25°に近いほど高スコア
  if(k==="batAngle"){
    const diff=Math.abs(n-m);
    return Math.min(100,Math.max(0,Math.round(100-diff*3)));
  }
  const r=n/m;
  return Math.min(100,Math.max(0,Math.round(k==="bodyFatPct"?(2-r)*70:r*70)));
}

// ── S〜Eグレード ──────────────────────────────────────
// S: 95+  A: 85+  B: 70+  C: 55+  D: 40+  E: <40
function grade(s){
  if(s===null)return null;
  if(s>=95)return"S";
  if(s>=85)return"A";
  if(s>=70)return"B";
  if(s>=55)return"C";
  if(s>=40)return"D";
  return"E";
}
const GRADE_COLOR={
  S:"#ffd700", // ゴールド
  A:"#22c55e", // 緑
  B:"#38bdf8", // 青
  C:"#f59e0b", // 黄
  D:"#fb923c", // オレンジ
  E:"#ef4444", // 赤
};
const GRADE_DESC={
  S:"エリート / Elite",
  A:"MLB水準 / MLB Level",
  B:"標準 / Average",
  C:"平均以下 / Below Avg",
  D:"要改善 / Needs Work",
  E:"要注意 / Critical",
};

function gradeColor(s){return s===null?"#64748b":GRADE_COLOR[grade(s)]||"#64748b";}
function gradeLabel(s){
  if(s===null)return"";
  const g=grade(s);
  return`<span style="background:${GRADE_COLOR[g]}22;color:${GRADE_COLOR[g]};border:1px solid ${GRADE_COLOR[g]}55;border-radius:4px;padding:1px 7px;font-size:11px;font-weight:900;font-family:monospace;">${g}</span>`;
}

function updateScore(k){
  const v=document.getElementById(k)?.value;
  const s=calcScore(k,v);
  const el=document.getElementById("sc-"+k);
  if(el)el.innerHTML=gradeLabel(s);
  const inp=document.getElementById(k);
  if(inp)inp.style.borderColor=s!==null?gradeColor(s)+"88":"#334155";
}

// ── 非対称性 ──────────────────────────────────────────
function calcAsym(l,r){
  const lv=parseFloat(l),rv=parseFloat(r);
  if(isNaN(lv)||isNaN(rv)||lv===0||rv===0)return null;
  return Math.round((1-Math.min(lv,rv)/Math.max(lv,rv))*100);
}
function asymColor(p){return p===null?"#64748b":p<=10?"#22c55e":p<=15?"#f59e0b":"#ef4444";}
function asymLabel(p){return p===null?"":p<=10?"Good":p<=15?"Caution":"Needs Work";}

function updateTrunkAsym(){
  const lv=document.getElementById("trunkL")?.value;
  const rv=document.getElementById("trunkR")?.value;
  const pct=calcAsym(lv,rv);
  const c=asymColor(pct);
  document.getElementById("trunk-asym-fill").style.width=(pct!==null?Math.min(pct*3,100):0)+"%";
  document.getElementById("trunk-asym-fill").style.background=c;
  document.getElementById("trunk-asym-pct").textContent=pct!==null?pct+"%":"—";
  document.getElementById("trunk-asym-pct").style.color=c;
  document.getElementById("trunk-asym-label").textContent=pct!==null?asymLabel(pct):"";
  document.getElementById("trunk-asym-label").style.color=c;
}

function updateTrunkScore(){

  }
}

// ── レーダーチャート ──────────────────────────────────
function radarSVG(D,size=220){
  const keys=["batSpeedKph","onPlane","batAngle","exitVelKph","hardHitPct","ffmi"];
  const lbls=["Bat SPD","On-Plane","Bat Angle","Exit Vel","Hard HIT","FFMI"];
  const cx=size/2,cy=size/2,r=size*0.33;
  const ang=keys.map((_,i)=>(Math.PI*2*i/keys.length)-Math.PI/2);
  const norm=(k,v)=>{const m=MLB[k]?.val;return(!m||!v)?0:Math.min(1.3,Math.max(0,parseFloat(v)/m));};
  const pPts=keys.map((k,i)=>{const n=norm(k,D[k]);return{x:cx+r*n*Math.cos(ang[i]),y:cy+r*n*Math.sin(ang[i])};});
  const mPts=ang.map(a=>({x:cx+r*Math.cos(a),y:cy+r*Math.sin(a)}));
  let s=`<svg width="${size}" height="${size}" style="overflow:visible;">`;
  [0.5,1,1.3].forEach(l=>{const pts=ang.map(a=>`${cx+r*l*Math.cos(a)},${cy+r*l*Math.sin(a)}`).join(" ");s+=`<polygon points="${pts}" fill="none" stroke="${l===1?"#334155aa":"#1e3a5f"}" stroke-width="${l===1?1.5:0.8}"/>`;});
  ang.forEach(a=>s+=`<line x1="${cx}" y1="${cy}" x2="${cx+r*1.3*Math.cos(a)}" y2="${cy+r*1.3*Math.sin(a)}" stroke="#1e3a5f" stroke-width="0.8"/>`);
  s+=`<polygon points="${mPts.map(p=>`${p.x},${p.y}`).join(" ")}" fill="#f59e0b10" stroke="#f59e0b" stroke-width="1.5" stroke-dasharray="5,3"/>`;
  s+=`<polygon points="${pPts.map(p=>`${p.x},${p.y}`).join(" ")}" fill="#38bdf828" stroke="#38bdf8" stroke-width="2"/>`;
  pPts.forEach(p=>s+=`<circle cx="${p.x}" cy="${p.y}" r="3.5" fill="#38bdf8"/>`);
  ang.forEach((a,i)=>{const lx=cx+(r*1.3+14)*Math.cos(a),ly=cy+(r*1.3+14)*Math.sin(a);s+=`<text x="${lx}" y="${ly}" text-anchor="middle" dominant-baseline="middle" style="font-size:8px;fill:#94a3b8;">${lbls[i]}</text>`;});
  return s+"</svg>";
}

// ── ドリル生成 ─────────────────────────────────────────
function buildDrills(D){

  const op=n("onPlane");if(op!==null&&op<70)drills.push({p:"HIGH",t:"On-Plane Efficiency — Pivot Drill",c:`On-Plane at ${op}% is below MLB standard (71%). Reset hip-first sequencing with 20 Pivot Drill swings per day.`});
  const ba=n("batAngle");if(ba!==null&&ba>-20)drills.push({p:"MED",t:"Bat Angle — Barrel Path Drill",c:`Bat angle at ${ba}° is above MLB average (-25°). Focus on keeping the barrel below the hands longer through the zone. Use tee drills with a focus on maintaining barrel angle through contact.`});
  const aa=n("attackAngle");if(aa!==null&&aa<5)drills.push({p:"MED",t:"Attack Angle — Tee Low-High Drill",c:`Attack angle at ${aa}° is too low. Raise tee height from hip to shoulder level to build upward swing feel. Target: +8 to +12°.`});
  const bf=n("bodyFatPct");if(bf!==null&&bf>18)drills.push({p:"MED",t:"Body Composition — Nutrition Management",c:`Body fat at ${bf}% exceeds MLB average (~15%). Optimize fat and carb timing, add aerobic work to reduce body fat and improve power-to-weight ratio.`});
  const la=n("launchAngle");if(la!==null&&la<12)drills.push({p:"MED",t:"Launch Angle — Elevated Tee Drill",c:`Launch angle at ${la}° is below the line-drive zone (15–24°). Set the tee high and repeat until the trajectory improves.`});
  const fm=n("ffmi");if(fm!==null&&fm<22.5)drills.push({p:"MED",t:"Lean Mass — Strength Training",c:`FFMI at ${fm} is below MLB reference (22.5). Focus on compound lifts (squat, deadlift, bench) 3–4x/week to build lean mass and power base.`});
  const tA=calcAsym(D.trunkL,D.trunkR);
  if(tA!==null&&tA>15)drills.push({p:"HIGH",t:"Trunk Rotation Asymmetry",c:`Trunk Rotation asymmetry is ${tA}% (target: under 10%). Use Pallof Press and cable rotations on the weaker side to close the gap — directly impacts rotational power at the plate.`});
  if(!drills.length)drills.push({p:"GOOD",t:"All Metrics at MLB Standard",c:"All metrics are within target range. Focus on game-situation batting to build consistency, then retest in 2–3 weeks."});
  return drills;
}
